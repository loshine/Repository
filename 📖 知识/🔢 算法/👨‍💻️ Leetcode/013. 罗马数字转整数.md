# 解题

从每个字母遍历到结束，发现如果全部相加，某些字母只需要再减去双倍就可以了。

```kotlin
class Solution {
    fun romanToInt(s: String): Int {
        var previous = 0
        var current: Int
        var total = 0
        var i = 0

        while (i < s.length) {
            current = when (s[i]) {
                'I' -> 1
                'V' -> 5
                'X' -> 10
                'L' -> 50
                'C' -> 100
                'D' -> 500
                'M' -> 1000
                else -> 0
            }
            if (current <= previous) {
                total += current
            } else {
                total = total - previous * 2 + current
            }
            previous = current
            i++
        }
        return total
    }
}
```
