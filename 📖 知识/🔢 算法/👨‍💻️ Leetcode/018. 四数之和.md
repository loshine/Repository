# 题目

给定一个包含 n 个整数的数组`nums`和一个目标值`target`，判断`nums`中是否存在四个元素 a，b，c 和 d ，使得 a + b + c + d 的值与`target`相等？找出所有满足条件且不重复的四元组。

# 解题

## 1. 递归法

使用递归将其简化为两数之和问题

```kotlin
class Solution {
    fun fourSum(nums: IntArray, target: Int): List<List<Int>> {
        nums.sort()
        return kSum(nums, 0, 4, target)
    }

    private fun kSum(nums: IntArray, start: Int, k: Int, target: Int): MutableList<MutableList<Int>> {
        val res = mutableListOf<MutableList<Int>>()
        if (k == 2) {
            var left = start
            var right = nums.size - 1
            while (left < right) {
                val sum = nums[left] + nums[right]
                when {
                    sum == target -> {
                        val path = ArrayList<Int>()
                        path.add(nums[left])
                        path.add(nums[right])
                        res.add(path)
                        while (left < right && nums[left] == nums[left + 1]) left++
                        while (left < right && nums[right] == nums[right - 1]) right--
                        left++
                        right--
                    }
                    sum < target -> left++
                    else -> right--
                }
            }
        } else {
            for (i in start until nums.size - (k - 1)) {
                if (i > start && nums[i] == nums[i - 1]) continue
                val temp = kSum(nums, i + 1, k - 1, target - nums[i])
                for (t in temp) {
                    t.add(0, nums[i])
                }
                res.addAll(temp)
            }
        }
        return res
    }
}
```

## 2. 简化为两数之和问题

加入特殊判断条件提升效率

```kotlin
class Solution {
    fun fourSum(nums: IntArray, target: Int): List<List<Int>> {
        val result = mutableListOf<MutableList<Int>>()
        val n = nums.size
        if (n < 4) return result
        nums.sort()
        for (i in 0 until n - 3) {
            if (i > 0 && nums[i] == nums[i - 1]) continue
            if (nums[i] + nums[i + 1] + nums[i + 2] + nums[i + 3] > target) break
            if (nums[i] + nums[n - 3] + nums[n - 2] + nums[n - 1] < target) continue
            for (j in i + 1 until n - 2) {
                if (j > i + 1 && nums[j] == nums[j - 1]) continue
                if (nums[i] + nums[j] + nums[j + 1] + nums[j + 2] > target) break
                if (nums[i] + nums[j] + nums[n - 2] + nums[n - 1] < target) continue
                var left = j + 1
                var right = n - 1

                while (left < right) {
                    val sum = nums[left] + nums[right] + nums[i] + nums[j]
                    when {
                        sum < target -> left++
                        sum > target -> right--
                        else -> {
                            result.add(mutableListOf(nums[i], nums[j], nums[left], nums[right]))
                            do {
                                left++
                            } while (nums[left] == nums[left - 1] && left < right)
                            do {
                                right--
                            } while (nums[right] == nums[right + 1] && left < right)
                        }
                    }
                }
            }
        }
        return result
    }
}
```
