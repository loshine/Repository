# 题目

给定一个只包括`(`，`)`，`{`，`}`，`[`，`]`的字符串，判断字符串是否有效。

有效字符串需满足：

1. 左括号必须用相同类型的右括号闭合。
2. 左括号必须以正确的顺序闭合。

> 注意空字符串可被认为是有效字符串。

# 解题

## 1. 栈

```kotlin
class Solution {
    fun isValid(s: String): Boolean {
        val stack: Stack<Char> = Stack()

        if (s.length % 2 == 1) return false
        if (s.isEmpty()) return true
        if (s[0] == ')' || s[0] == '}' || s[0] == ']') return false
        for (i in 0 until s.length) {
            if (s[i] == '(' || s[i] == '{' || s[i] == '[') {
                stack.push(s[i])
            } else if (s[i] == ')' && stack[stack.lastIndex] == '(') {
                stack.pop()
            } else if (s[i] == '}' && stack[stack.lastIndex] == '{') {
                stack.pop()
            } else if (s[i] == ']' && stack[stack.lastIndex] == '[') {
                stack.pop()
            } else {
                return false
            }
        }

        return stack.isEmpty()
    }
}
```

## 2. 双向队列

```kotlin
class Solution {
    fun isValid(s: String): Boolean {
        val stack = ArrayDeque<Char>()
        for (c in s) {
            when (c) {
                '(', '{', '[' -> {
                    stack.addFirst(c)
                }
                else -> {
                    when (c) {
                        ')' -> {
                            if (stack.poll() != '(') return false
                        }
                        '}' -> {
                            if (stack.poll() != '{') return false
                        }
                        ']' -> {
                            if (stack.poll() != '[') return false
                        }
                    }
                }
            }
        }
        return stack.isEmpty()
    }
}
```
